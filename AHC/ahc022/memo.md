$$ 
\begin{align*}
        &\sum_{i = 1}^N (x_i - \overline{x})^2\\
    =& \sum_{i=1}^N(x_i^2 - 2x_i\overline{x} + \overline{x}^2)\\
    =&\sum_{i=1}^N x_i^2 - 2\overline{x} \sum x_i + N\overline{x}^2\\
    = &\sum x_i^2 - N \overline{x}^2\\
    = &\sum x_i^2 -N (1/N)^2(\sum x_i)^2\\
    = &\sum x_i^2 - (1/N)(\sum x_i)^2
\end{align*}
$$
